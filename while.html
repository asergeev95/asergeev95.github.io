#include <stdio.h>
void main()
{
    int s, n;
    s = 47;
    n = 1;
    while (s > 0) {
        s = s – 9;
        n = n + 4;
    }
    printf("%d\n", n);
}
Цикл while вы­пол­ня­ет­ся до тех пор, пока ис­тин­но усло­вие s > 0, т. е. пе­ре­мен­ная s определяет, сколь­ко раз вы­пол­нит­ся цикл. По­сколь­ку из­на­чаль­но s = 47, цикл вы­пол­нит­ся 6 раз, следовательно, n = 6 · 4 + 1 = 25. Ответ: 25.
